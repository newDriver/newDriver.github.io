面向对象的关术语

字面量法： 声明某种数据结构的一种常用的简化写法，这种形式在很多编程语言中都适用。

//声明一个数组
var arr1 = new Array("a", "b", "c");
//声明一个数组字面量语法
var arr2 = ["a", "b", "c"];
//声明一个字符串
var str1 = new String("abc");
//声明一个字符串字面量语法
var str2 = "abc";
通过typeof 运算符来验证是否是对象类型

自定义对象

自定义对象是一种比较特殊的数据结构，可以用来描述我们生活中的某种具体的一个事物，该结构内可以存储其他复杂的数据结构，具有强大的封装性。
//声明一个自定义对象
var obj = new Object();
console.log(typeof obj);
//声明一个对象字面量
var obj1 = {};
console.log(obj1);
console.log(typeof obj1);

声明一个对象字面量 带属性值

var obj2 = {
    name : "王亚杰"  ,
    sex : '男',
    say : function (sayContent) {
        //this指针，在这里指向的是当前对象（也就是this代表obj3）

        console.log("大家好，我叫：" + this.name + "，性别：" + this.sex + ", 简单解释一下：" + sayContent);
    }
};
console.log(obj2);
obj3.sayHello("这就是对象字面量");
可以通过点语法"."或中括号"[]"来访问属性。
当明确的知道我们要访问的属性的名字的时候，用以上两种方式都可以，
当不明确要访问的属性名字的时候，如形参或者变量，需要用中括号的形式进行访问。

工厂函数

工厂函数创建对象，所谓工厂函数就是一个函数，我们通过这个函数传参，根据参数来生成对象，这种操作就像一个工厂的车间一样
// 工厂函数模拟类的概念
function makePerson(name,age){
    //对象也是可以作为函数返回值的
    return {
        name:name,
        // 上边name 是形参 下边函数内是属性
        age: age ,
        sayHellow:function(sayContent){
            console.log("大家好！我是"+this.name+"今年"+this.age+"，我要说:"+sayContent)
            //上边sayContent是形参 传实参进入后 带入输出实参

        }
    };
    // var obj = {};
    // return obj ;
}

// 创建一个人的对象
var person1 = makePerson("狗蛋",18);
person1.sayHellow("好神奇啊");
// 在创建一个
var person2 = makePerson("二狗",15);
person2.sayHellow("就是！就是！");

//判断是否相等
console.log(person2===person1);
console.log(typeof person1);
console.log(typeof person2);

实现方法简单，在函数内创建一个对象，给对象赋予属性及方法再将对象返回即可，
不足之处工厂模式不能识别对象的类型，typeof 类型都是Object

构造函数：构造函数就是一个普通函数，只不过我们习惯把构造函数的名字首字母大写（程序员之间的一种约定） 

// 构造函数模拟类的概念
function Person(name,age) {
    // 声明属性
    // this指针表示通过该构造函数创建出来的那个对象，也就是我们经常说的当前对象
    this.name = name;  //前边的是属性 后边是属性值 出入的参数
    this.age = age;
    this.sayHello = function(sayContent){
        console.log("大家好！我是"+this.name+"，今年"+this.age+"我要说"+sayContent);
    }
    //当使用NEW关键字配合构造函数的时候，会默认省略了return this;操作，就是返回当前对象的操作
    // return this;
}

//通过构造函数创建一个人的对象
var person1 = new Person("狗蛋",18);
person1.sayHello("我要用尽洪荒之力来学习！");

var person2 = new Person("二虎",15);
person2.sayHello("狗蛋你变了");

var person3 = person2;  //依然只有person1 person2两个对象，与person2指向同一地址，但是没有生成新的对象
person3.sayHello("对对对");

person1 = new Person("大猫",23); //生成一个新的对象
person1.sayHello("我是阿猫")
console.log("Person1"+person1.name); //大猫
console.log("Person2"+person2.name); //二虎
console.log("Person3"+person3.name); //二虎

person3.name = "麻匪"; //改变了内存地址里的内容
console.log("Person1"+person1.name); //大猫
console.log("Person2"+person2.name); //麻匪
console.log("Person3"+person3.name); //麻匪
//判断person1 person2是否是指向同一个对象
console.log(person1 === person2); //false
//判断person3 person2是否是指向同一个对象

console.log(person3 === person2); //true

console.log(typeof person1);
console.log(typeof person2);

//通过instanceof 运算符可以判断某个变量指针是否是某个构造函数创建出来的
console.log(person1 instanceof Person);//true
console.log(person2 instanceof Person); //true

// object是所有构造函数的根
console.log(person3 instanceof Object); //true
console.log(person1 instanceof String); //false

}

创建新的实例经历了以下的过程

不足之处每次创建实例的时候都要重新创建一次方法（理论上每次创建对象的时候对象的属性均不同，而对象的方法是相同的），然而创建两次完全相同的方法是没有必要的，虽然我们可以使用下面的方法处理这个问题，将函数移到对象外面

function Person(name,age) {
    // 声明属性
    // this指针表示通过该构造函数创建出来的那个对象，也就是我们经常说的当前对象
    this.name = name;  //前边的是属性 后边是属性值 出入的参数
    this.age = age;
    this.sayHello = sayHello;
}
function sayHello (sayContent){}{
    console.log("大家好！我是"+this.name+"，今年"+this.age+"我要说"+this.sayContent);
}

将sayHello设置成全局函数 这样新生成的实例访问的都是同一个函数。
但在全局作用域中定义了一个实际仅预提供给Person使用的函数，在全局作用域中定义了仅供特定对象使用的方法，当要处理的方法量较多时 这个解决方案就和要实现的封装相悖。这里可以通过原型来解决此问题。
原型模式
通过函数的prototype（原型）属性，实现属性和方法的共享
function person(name,age){}
person.prototype.name=name;
person.prototype.age=age;
person.prototype.sayHello=function(sayContent){
console.log(“大家好！我是”+ this.name+”，今年”+ this.age+”我要说”+sayContent);
}

var person1=new person();
var person2=new person();
person1.name='test1';

person1.say("大家好");
person2.say("hellow");

不足之处省略了构造函数传递初始化参数这一环节，所有实例均在默认情况下取得了相同的属性值，每次都需要手动修改。由于共享，当生成的一个实例进行修改引用时，其他的生成的实例的属性和方法也会随之更改。因此一般不会单独使用原型模式。就有了下面的混合模式

混合模式构造函数+创建原型模式创建对象

function Person (name) {
    this.name = name ;
    this.say = function(){
        console.log("东风吹，战鼓擂，我是"+this.name+"，我怕谁！");
    }
}
//创建两个人的对象
var person1 = new Person("狗蛋");
var person2 = new Person("二虎");

console.log(person1.say == person2.say); //比的是地址 false 因为函数也是对象，也需要占用堆内存，每次对象调用函数的时候都会在对内存开辟一块空间，如果频繁调用会浪费空间，消耗性能 
/*
原型模式创建对象
每个构造函数都有一个原型属性（prototype),这个原型属性指向一个原型对象。我们可以给这个原型对象动态添加属性和方法，这些属性和方法，我们通过该构造函数所创建的对象，都可以使用，而且可以达到属性和方法共享的目的。

*/
function Dog (age){
    this.age = age;
}

//给dog构造函数 的原型对象添加一个say方法，那么通过Dog构造函数所创建出来的一个个实例对象，都会有这个say方法，并且是共享的。
Dog.prototype.say =function(){
    // this指针在这里和原来一样，指向通过该构造函数创建处理 的对象，也就是当前对象
    console.log("年龄"+this.age);
}
// class_name.prototype.method_name = function(first_argument) {
//     // body...
// };
// 创建两只狗狗的对象
var dog1 = new Dog(18);
var dog2 = new Dog(20);
console.log(dog1.say == dog2.say);//true
dog1.say();//年龄18
dog2.say();//年龄20

混合模式中使用构造函数用于定义实例的属性，使用原型模式用于定义方法和一些共享的属性。每个实例都会有自己的属性，但同时又共享着方法，最大限度的节省了内存。另外这种模式还支持传递初始参数。
